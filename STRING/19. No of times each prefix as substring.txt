constexpr ll mod = 1e9 + 7;
const ll N=3e6+5;
ll pre[N];

vector<int> z_function(string s) {
    int n = (int) s.length();
    vector<int> z(n);
    for (int i = 1, l = 0, r = 0; i < n; ++i) {
        if (i <= r)
            z[i] = min (r - i + 1, z[i - l]);
        while (i + z[i] < n && s[z[i]] == s[i + z[i]])
            ++z[i];
        if (i + z[i] - 1 > r)
            l = i, r = i + z[i] - 1;
    }
    return z;
}

void solve(){
   ll n;
   string s;
   cin>>s;
   n= s.size();
   vector<int> v = z_function(s);
   
   //for(ll i=0;i<=n;i++) pre[i]=1;
   for(ll i=0;i<n;i++){
     pre[v[i]]++;
   }
   for(ll i=n-1;i>=1;i--) pre[i]+=pre[i+1];
   
   for(ll i=1;i<=n;i++)
    cout<<pre[i]+1<<' ';
   cout<<endl;


}
